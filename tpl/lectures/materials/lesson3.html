Javascript

Что такое Javascript? И зачем он нужен

  • JavaScript - язык, который делает web-странички «живыми».
  • JS-программа - это скрипт.
  • Cкрипт подключаются к HTML-странице через <script>
  • Скрипт выполняется сразу же, как загрузиласьстраница

Что умеет JavaScript?

Современный JavaScript – это «безопасный» язык программирования общего назначения. Он не предоставляет низкоуровневых средств работы с памятью, процессором, так как изначально был ориентирован на браузеры, в которых это не требуется.

  • Создавать новые HTML-теги, удалять существующие, менять стили элементов, прятать, показывать элементы и т.п.
  • Реагировать на действия посетителя, обрабатывать клики мыши, перемещения курсора, нажатия на клавиатуру и т.п.
  • Посылать запросы на сервер и загружать данные без перезагрузки страницы (эта технология называется "AJAX").
  • Получать и устанавливать cookie, запрашивать данные, выводить сообщения…
  • и многое другое...

Преимущества JavaScript

  • Полная интеграция с HTML/CSS.
  • Простые вещи делаются просто.
  • Поддерживается всеми распространёнными браузерами и включён по умолчанию.

Переменные

function printValues() {
    var message = "Я - строка"; // строка
    var number_4 = 4; // число
    var always_true = true; // логическая переменная
    var MESSAGE = "Я - другая строка"; /* message и MESSAGE - две разные переменные*/
    var hello = 'привет', five = 5; /* можно объявлять переменные и так*/
    alert("message - " + message);
    alert("always_true - " + always_true);
    alert("number_4 - " + number_4);
    alert("MESSAGE - " + MESSAGE);
    alert("hello + five = " + hello + five);
}

Циклы, функции и аргументы

function range(number) {
    var range_list = []
    var i;
    for (i = 1; i <= number; i++) {
        range_list.push(i);
    }
    return range_list;
}
function printRange() {
    var number = parseInt(document.getElementById("number").value);
    if (number && number > 0) {
        alert(range(number));
        return;
    }
    alert("Нужно ввести положительное число от 1 до N");
}
